電子を水に打ち込む

  • 幅100cm、長さ400cmの直方体の水(cell=1)に、左から 1 GeV の電子を1発打ち込んでみた。

インプットファイル

[ Parameters ]
  maxcas = 1
  maxbch = 1

  negs = 2

$======================================
[ source ]
   s-type = 1             $ mono-energetic axial source
     proj = electron
       z0 = -100
       z1 = -100
       e0 = 1000

[ surface ]

 99  so    2000
